116    pub(crate) fn get_or_create_type<K: Into<Arc<str>>>(
117        &mut self,
118        key: K,
119        type_ref: TypeRef,
120    ) -> BranchPtr {
121        let key = key.into();
122        match self.types.entry(key.clone()) {
123            Entry::Occupied(mut e) => {
124                let mut branch = BranchPtr::from(e.get());
125                branch.repair_type_ref(type_ref);
126                branch
127            }
128            Entry::Vacant(e) => {
129                let mut branch = Branch::new(type_ref);
130                let mut branch_ref = BranchPtr::from(&mut branch);
131                branch_ref.name = Some(key);
132                e.insert(branch);
133                branch_ref
134            }
135        }
136    }

138    /// Encodes all changes from current transaction block store up to a given `snapshot`.
139    /// This enables to encode state of a document at some specific point in the past.
140    pub fn encode_state_from_snapshot<E: Encoder>(
141        &self,
142        snapshot: &Snapshot,
143        encoder: &mut E,
144    ) -> Result<(), Error> {
145        if !self.skip_gc {
146            return Err(Error::Gc);

148        self.write_blocks_to(&snapshot.state_map, encoder);
149        snapshot.delete_set.encode(encoder);

154    pub(crate) fn write_blocks_to<E: Encoder>(&self, sv: &StateVector, encoder: &mut E) {
155        let local_sv = self.blocks.get_state_vector();
156        let mut diff = Vec::with_capacity(sv.len());
157        for (&client_id, &clock) in sv.iter() {
158            if local_sv.contains_client(&client_id) {
159                diff.push((client_id, clock.min(local_sv.get(&client_id))));
160            }
161        }
162        // Write items with higher client ids first
163        // This heavily improves the conflict algorithm.
164        diff.sort_by(|a, b| b.0.cmp(&a.0));
165
166        encoder.write_var(diff.len());
167        for (client, clock) in diff {
168            let blocks = self.blocks.get_client(&client).unwrap();
169            let clock = clock.min(blocks.clock() + 1);
170            let last_idx = blocks.find_pivot(clock - 1).unwrap();
171            // write # encoded structs
172            encoder.write_var(last_idx + 1);
173            encoder.write_client(client);
174            encoder.write_var(0);
175            for i in 0..last_idx {
176                let block = blocks[i].as_slice();
177                block.encode(encoder);
178            }
179            let last_block = &blocks[last_idx];
180            // write first struct with an offset
181            let mut slice = last_block.as_slice();
182            slice.trim_end(slice.clock_end() - (clock - 1));
183            slice.encode(encoder);
184        }
185    }

187    /// Compute a diff to sync with another client.
188    ///
189    /// This is the most efficient method to sync with another client by only
190    /// syncing the differences.
191    ///
192    /// The sync protocol in Yrs/js is:
193    /// * Send StateVector to the other client.
194    /// * The other client comutes a minimal diff to sync by using the StateVector.
195    pub fn encode_diff<E: Encoder>(&self, sv: &StateVector, encoder: &mut E) {
196        //TODO: this could be actually 2 steps:
197        // 1. create Diff of block store and remote state vector (it can have lifetime of bock store)
198        // 2. make Diff implement Encode trait and encode it
199        // this way we can add some extra utility method on top of Diff (like introspection) without need of decoding it.
200        self.write_blocks_from(sv, encoder);
201        let delete_set = DeleteSet::from(&self.blocks);
202        delete_set.encode(encoder);
203    }

205    pub(crate) fn write_blocks_from<E: Encoder>(&self, sv: &StateVector, encoder: &mut E) {
206        let local_sv = self.blocks.get_state_vector();
207        let mut diff = Self::diff_state_vectors(&local_sv, sv);
208
209        // Write items with higher client ids first
210        // This heavily improves the conflict algorithm.
211        diff.sort_by(|a, b| b.0.cmp(&a.0));
212
213        encoder.write_var(diff.len());
214        for (client, clock) in diff {
215            let blocks = self.blocks.get_client(&client).unwrap();
216            let clock = clock.max(blocks.get(0).map(|i| i.clock_start()).unwrap_or_default()); // make sure the first id exists
217            let start = blocks.find_pivot(clock).unwrap();
218            // write # encoded structs
219            encoder.write_var(blocks.len() - start);
220            encoder.write_client(client);
221            encoder.write_var(clock);
222            let first_block = blocks.get(start).unwrap();
223            // write first struct with an offset
224            let offset = clock - first_block.clock_start();
225            let mut slice = first_block.as_slice();
226            slice.trim_start(offset);
227            slice.encode(encoder);
228            for i in (start + 1)..blocks.len() {
229                let block = &blocks[i];
230                block.as_slice().encode(encoder);
231            }
232        }
233    }

235    fn diff_state_vectors(local_sv: &StateVector, remote_sv: &StateVector) -> Vec<(ClientID, u32)> {
236        let mut diff = Vec::new();
237        for (client, &remote_clock) in remote_sv.iter() {
238            let local_clock = local_sv.get(client);
239            if local_clock > remote_clock {
240                diff.push((*client, remote_clock));
241            }
242        }
243        for (client, _) in local_sv.iter() {
244            if remote_sv.get(client) == 0 {
245                diff.push((*client, 0));
246            }
247        }
248        diff
249    }

251    pub fn get_type_from_path(&self, path: &Path) -> Option<BranchPtr> {
252        let mut i = path.iter();
253        if let Some(PathSegment::Key(root_name)) = i.next() {
254            let mut current = self.get_type(root_name.clone())?;
255            while let Some(segment) = i.next() {
256                match segment {
257                    PathSegment::Key(key) => {
258                        let child = current.map.get(key)?;
259                        if let ItemContent::Type(child_branch) = &child.content {
260                            current = BranchPtr::from(child_branch.as_ref());
261                        } else {
262                            return None;
263                        }
264                    }
265                    PathSegment::Index(index) => {
266                        if let Some((ItemContent::Type(child_branch), _)) = current.get_at(*index) {
267                            current = child_branch.into();
268                        } else {
269                            return None;
270                        }
271                    }
272                }
273            }
274            Some(current)
275        } else {
276            None
277        }
278    }

280    /// Consumes current block slice view, materializing it into actual block representation equivalent,
281    /// splitting underlying block along [ItemSlice::start]/[ItemSlice::end] offsets.
282    ///
283    /// Returns a block created this way, that represents the boundaries that current [ItemSlice]
284    /// was representing.
285    pub(crate) fn materialize(&mut self, mut slice: ItemSlice) -> ItemPtr {
286        let id = slice.id().clone();
287        let blocks = self.blocks.get_client_mut(&id.client).unwrap();
288        let mut links = None;
289        let item = slice.ptr.deref();
290        if item.info.is_linked() {
291            links = self.linked_by.get(&slice.ptr).cloned();
292        }
293
294        let mut index = None;
295        let mut ptr = if slice.adjacent_left() {
296            slice.ptr
297        } else {
298            let mut i = blocks.find_pivot(id.clock).unwrap();
299            if let Some(new) = slice.ptr.splice(slice.start, OffsetKind::Utf16) {
300                if let Some(source) = links.clone() {
301                    let dest = self.linked_by.entry(ItemPtr::from(&new)).or_default();
302                    dest.extend(source);
303                }
304                blocks.insert(i + 1, BlockCell::Block(new));
305                i += 1;
306                //todo: txn merge blocks insert?
307                index = Some(i);
308            }
309            let ptr = blocks[i].as_item().unwrap();
310            slice = ItemSlice::new(ptr, 0, slice.end - slice.start);
311            ptr
312        };
313
314        if !slice.adjacent_right() {
315            // split block on the right side
316            let i = if let Some(i) = index {
317                i
318            } else {
319                let last_id = slice.last_id();
320                blocks.find_pivot(last_id.clock).unwrap()
321            };
322            let new = ptr.splice(slice.len(), OffsetKind::Utf16).unwrap();
323            if let Some(source) = links {
324                let dest = self.linked_by.entry(ItemPtr::from(&new)).or_default();
325                dest.extend(source);
326            }
327            blocks.insert(i + 1, BlockCell::Block(new));
328            //todo: txn merge blocks insert?
329        }
330
331        ptr
332    }

345    pub(crate) fn follow_redone(&self, id: &ID) -> Option<ItemSlice> {
346        let mut next_id = Some(*id);
347        let mut slice = None;
348        while let Some(next) = next_id.as_mut() {
349            slice = self.blocks.get_item_clean_start(next);
350            if let Some(slice) = &slice {
351                next_id = slice.ptr.redone;
352            } else {
353                break;
354            }
355        }
356        slice
357    }

360impl Encode for Store {
361    /// Encodes the document state to a binary format.
362    ///
363    /// Document updates are idempotent and commutative. Caveats:
364    /// * It doesn't matter in which order document updates are applied.
365    /// * As long as all clients receive the same document updates, all clients
366    ///   end up with the same content.
367    /// * Even if an update contains known information, the unknown information
368    ///   is extracted and integrated into the document structure.
369    fn encode<E: Encoder>(&self, encoder: &mut E) {
370        self.encode_diff(&StateVector::default(), encoder)
371    }
372}

538pub struct StoreEvents {

540    /// newest updates once they are committed and compacted.
541    pub transaction_cleanup_events: Observer<TransactionCleanupFn>,

543    /// Handles subscriptions for the `afterTransactionCleanup` event. Events are called with the

546
547    /// A subscription handler. It contains all callbacks with registered by user functions that
548    /// are supposed to be called, once a new update arrives.

552    /// are supposed to be called, once a new update arrives.
553    pub update_v2_events: Observer<UpdateFn>,
554
555    /// Handles subscriptions for subdocs events.
556    pub subdocs_events: Observer<SubdocsFn>,

558    pub destroy_events: Observer<DestroyFn>,

561impl StoreEvents {
562    pub fn emit_update_v1(&self, txn: &TransactionMut) {
563        if self.update_v1_events.has_subscribers() {
564            if !txn.delete_set.is_empty() || txn.after_state != txn.before_state {
565                // produce update only if anything changed
566                let update = UpdateEvent::new_v1(txn);
567                self.update_v1_events
568                    .trigger(|callback| callback(txn, &update));
569            }
570        }
571    }
572
573    pub fn emit_update_v2(&self, txn: &TransactionMut) {
574        if self.update_v2_events.has_subscribers() {
575            if !txn.delete_set.is_empty() || txn.after_state != txn.before_state {
576                // produce update only if anything changed
577                let update = UpdateEvent::new_v2(txn);
578                self.update_v2_events.trigger(|fun| fun(txn, &update));
579            }
580        }
581    }
582
583    pub fn emit_after_transaction(&self, txn: &mut TransactionMut) {
584        self.after_transaction_events.trigger(|fun| fun(txn));
585    }
586
587    pub fn emit_transaction_cleanup(&self, txn: &TransactionMut) {
588        if self.transaction_cleanup_events.has_subscribers() {
589            let event = TransactionCleanupEvent::new(txn);
590            self.transaction_cleanup_events
591                .trigger(|fun| fun(txn, &event));
592        }
593    }
