28/// Trait defining read capabilities present in a transaction. Implemented by both lightweight
29/// [read-only](Transaction) and [read-write](TransactionMut) transactions.
30pub trait ReadTxn: Sized {
31    fn store(&self) -> &Store;
32
33    /// Returns state vector describing current state of the updates.
34    fn state_vector(&self) -> StateVector {
35        self.store().blocks.get_state_vector()
36    }
37
38    /// Returns a snapshot which describes a current state of updates and removals made within
39    /// the corresponding document.
40    fn snapshot(&self) -> Snapshot {
41        let store = self.store();
42        let blocks = &store.blocks;
43        let sv = blocks.get_state_vector();
44        let ds = DeleteSet::from(blocks);
45        Snapshot::new(sv, ds)
46    }
47
48    /// Encodes all changes from current transaction block store up to a given `snapshot`.
49    /// This enables to encode state of a document at some specific point in the past.
50    fn encode_state_from_snapshot<E: Encoder>(
51        &self,
52        snapshot: &Snapshot,
53        encoder: &mut E,
54    ) -> Result<(), Error> {
55        self.store().encode_state_from_snapshot(snapshot, encoder)
56    }
57
58    /// Encodes the difference between remove peer state given its `state_vector` and the state
59    /// of a current local peer
60    fn encode_diff<E: Encoder>(&self, state_vector: &StateVector, encoder: &mut E) {
61        self.store().encode_diff(state_vector, encoder)
62    }
63
64    fn encode_diff_v1(&self, state_vector: &StateVector) -> Vec<u8> {
65        let mut encoder = EncoderV1::new();
66        self.encode_diff(state_vector, &mut encoder);
67        encoder.to_vec()
68    }
69
70    fn encode_diff_v2(&self, state_vector: &StateVector) -> Vec<u8> {
71        let mut encoder = EncoderV2::new();
72        self.encode_diff(state_vector, &mut encoder);
73        encoder.to_vec()
74    }
75
76    fn encode_state_as_update<E: Encoder>(&self, sv: &StateVector, encoder: &mut E) {
77        let store = self.store();
78        store.write_blocks_from(sv, encoder);
79        let ds = DeleteSet::from(&store.blocks);
80        ds.encode(encoder);
81    }
82
83    fn encode_state_as_update_v1(&self, sv: &StateVector) -> Vec<u8> {
84        let mut encoder = EncoderV1::new();
85        self.encode_state_as_update(sv, &mut encoder);
86        // check for pending data
87        merge_pending_v1(encoder.to_vec(), self.store())
88    }
89
90    fn encode_state_as_update_v2(&self, sv: &StateVector) -> Vec<u8> {
91        let mut encoder = EncoderV2::new();
92        self.encode_state_as_update(sv, &mut encoder);
93
94        // check for pending data
95        merge_pending_v2(encoder.to_vec(), self.store())
96    }

98    /// Returns an iterator over top level (root) shared types available in current [Doc].
99    fn root_refs(&self) -> RootRefs {
100        let store = self.store();
101        RootRefs(store.types.iter())
102    }
103
104    /// Returns a collection of globally unique identifiers of sub documents linked within

111    /// Returns a collection of sub documents linked within the structures of this document store.

115    }

117    /// Returns a [TextRef] data structure stored under a given `name`. Text structures are used for
118    /// collaborative text editing: they expose operations to append and remove chunks of text,
119    /// which are free to execute concurrently by multiple peers over remote boundaries.
120    ///
121    /// If not structure under defined `name` existed before, [None] will be returned.
122    ///
123    /// If a structure under defined `name` already existed, but its type was different it will be
124    /// reinterpreted as a text (in such case a sequence component of complex data type will be
125    /// interpreted as a list of text chunks).
126    #[inline]
127    fn get_text<N: Into<Arc<str>>>(&self, name: N) -> Option<TextRef> {
128        TextRef::root(name).get(self)
129    }
130
131    /// Returns an [ArrayRef] data structure stored under a given `name`. Array structures are used for
132    /// storing a sequences of elements in ordered manner, positioning given element accordingly
133    /// to its index.
134    ///
135    /// If not structure under defined `name` existed before, [None] will be returned.
136    ///
137    /// If a structure under defined `name` already existed, but its type was different it will be
138    /// reinterpreted as an array (in such case a sequence component of complex data type will be
139    /// interpreted as a list of inserted values).
140    #[inline]
141    fn get_array<N: Into<Arc<str>>>(&self, name: N) -> Option<ArrayRef> {
142        ArrayRef::root(name).get(self)
143    }
144
145    /// Returns a [MapRef] data structure stored under a given `name`. Maps are used to store key-value
146    /// pairs associated. These values can be primitive data (similar but not limited to
147    /// a JavaScript Object Notation) as well as other shared types (Yrs maps, arrays, text
148    /// structures etc.), enabling to construct a complex recursive tree structures.
149    ///
150    /// If not structure under defined `name` existed before, [None] will be returned.
151    ///
152    /// If a structure under defined `name` already existed, but its type was different it will be
153    /// reinterpreted as a map (in such case a map component of complex data type will be
154    /// interpreted as native map).
155    #[inline]
156    fn get_map<N: Into<Arc<str>>>(&self, name: N) -> Option<MapRef> {
157        MapRef::root(name).get(self)
158    }
159
160    /// Returns a [XmlFragmentRef] data structure stored under a given `name`. XML elements represent
161    /// nodes of XML document. They can contain attributes (key-value pairs, both of string type)
162    /// and other nested XML elements or text values, which are stored in their insertion
163    /// order.
164    ///
165    /// If not structure under defined `name` existed before, [None] will be returned.
166    ///
167    /// If a structure under defined `name` already existed, but its type was different it will be
168    /// reinterpreted as a XML element (in such case a map component of complex data type will be
169    /// interpreted as map of its attributes, while a sequence component - as a list of its child
170    /// XML nodes).
171    #[inline]
172    fn get_xml_fragment<N: Into<Arc<str>>>(&self, name: N) -> Option<XmlFragmentRef> {
173        XmlFragmentRef::root(name).get(self)
174    }
175
176    /// If current document has been inserted as a sub-document, returns a reference to a parent
177    /// document, which contains it.
178    fn parent_doc(&self) -> Option<Doc> {
179        if let Some(item) = self.store().parent.as_deref() {
180            if let ItemContent::Doc(parent_doc, _) = &item.content {
181                return parent_doc.clone();
182            }
183        }
184
185        None
186    }
187
188    /// If current document has been inserted as a sub-document, returns its [BranchID].
189    fn branch_id(&self) -> Option<BranchID> {
190        if let Some(item) = self.store().parent {
191            Some(BranchID::Nested(item.id))
192        } else {
193            None
194        }
195    }

262fn merge_pending_v1(update: Vec<u8>, store: &Store) -> Vec<u8> {
263    let mut merge = VecDeque::new();
264    if let Some(pending) = store.pending.as_ref() {
265        merge.push_back(pending.update.encode_v1());
266    }
267    if let Some(pending_ds) = store.pending_ds.as_ref() {
268        let mut u = Update::new();
269        u.delete_set = pending_ds.clone();
270        merge.push_back(u.encode_v1());
271    }
272    if merge.is_empty() {
273        update
274    } else {
275        merge.push_front(update);
276        merge_updates_v1(merge).unwrap()
277    }
278}
279
280fn merge_pending_v2(update: Vec<u8>, store: &Store) -> Vec<u8> {
281    let mut merge = VecDeque::new();
282    if let Some(pending) = store.pending.as_ref() {

284    }

289    }

291        update

295    }

320/// Read-write transaction. It can be used to modify an underlying state of the corresponding [Doc].
321/// Read-write transactions require an exclusive access to document store - only one such
322/// transaction can be present per [Doc] at the same time (read-only [Transaction]s are not allowed
323/// to coexists at the same time as well).
324///
325/// This transaction type stores the information about all of the changes performed in its scope.
326/// These will be used during [TransactionMut::commit] call to optimize metadata of incoming updates,
327/// triggering necessary event callbacks etc. For performance reasons it's preferred to batch as
328/// many updates as possible using the same transaction.
329///
330/// In Yrs transactions are always auto-committing all of their changes when dropped. Rollbacks are
331/// not supported (if some operations needs to be undone, this can be achieved using [UndoManager])
332pub struct TransactionMut<'doc> {
333    pub(crate) store: RwLockWriteGuard<'doc, Store>,
334    /// State vector of a current transaction at the moment of its creation.
335    pub(crate) before_state: StateVector,
336    /// Current state vector of a transaction, which includes all performed updates.
337    pub(crate) after_state: StateVector,
338    /// ID's of the blocks to be merged.
339    pub(crate) merge_blocks: Vec<ID>,
340    /// Describes the set of deleted items by ids.
341    pub(crate) delete_set: DeleteSet,
342    /// We store the reference that last moved an item. This is needed to compute the delta
343    /// when multiple ContentMove move the same item.
344    pub(crate) prev_moved: HashMap<ItemPtr, ItemPtr>,
345    /// All types that were directly modified (property added or child inserted/deleted).
346    /// New types are not included in this Set.
347    pub(crate) changed: HashMap<TypePtr, HashSet<Option<Arc<str>>>>,
348    pub(crate) changed_parent_types: Vec<BranchPtr>,
349    pub(crate) subdocs: Option<Box<Subdocs>>,
350    pub(crate) origin: Option<Origin>,
351    doc: Doc,
352    committed: bool,
353}

493    /// Applies given `id_set` onto current transaction to run multi-range deletion.
494    /// Returns a remaining of original ID set, that couldn't be applied.
495    pub(crate) fn apply_delete(&mut self, ds: &DeleteSet) -> Option<DeleteSet> {
496        let mut unapplied = DeleteSet::new();
497        for (client, ranges) in ds.iter() {
498            if let Some(mut blocks) = self.store.blocks.get_client_mut(client) {
499                let state = blocks.clock();
500
501                for range in ranges.iter() {
502                    let clock = range.start;
503                    let clock_end = range.end;
504
505                    if clock < state {
506                        if state < clock_end {
507                            unapplied.insert(ID::new(*client, clock), clock_end - state);
508                        }
509                        // We can ignore the case of GC and Delete structs, because we are going to skip them
510                        if let Some(mut index) = blocks.find_pivot(clock) {
511                            // We can ignore the case of GC and Delete structs, because we are going to skip them
512                            let ptr = &mut blocks[index];
513                            if let Some(item) = ptr.as_item() {
514                                // split the first item if necessary
515                                if !item.is_deleted() && item.id.clock < clock {
516                                    if let Some(split) = self
517                                        .store
518                                        .blocks
519                                        .split_block_inner(item, clock - item.id.clock)
520                                    {
521                                        if item.moved.is_some() {
522                                            if let Some(&prev_moved) = self.prev_moved.get(&item) {
523                                                self.prev_moved.insert(split, prev_moved);
524                                            }
525                                        }
526
527                                        index += 1;
528                                        self.merge_blocks.push(*split.id());
529                                    }
530                                    blocks = self.store.blocks.get_client_mut(client).unwrap();
531                                }
532
533                                while index < blocks.len() {
534                                    let block = &mut blocks[index];
535                                    if let Some(item) = block.as_item() {
536                                        if item.id.clock < clock_end {
537                                            if !item.is_deleted() {
538                                                if item.id.clock + item.len() > clock_end {
539                                                    if let Some(split) =
540                                                        self.store.blocks.split_block_inner(
541                                                            item,
542                                                            clock_end - item.id.clock,
543                                                        )
544                                                    {
545                                                        if item.moved.is_some() {
546                                                            if let Some(&prev_moved) =
547                                                                self.prev_moved.get(&item)
548                                                            {
549                                                                self.prev_moved
550                                                                    .insert(split, prev_moved);
551                                                            }
552                                                        }
553                                                        if item.info.is_linked() {
554                                                            if let Some(links) = self
555                                                                .store
556                                                                .linked_by
557                                                                .get(&item)
558                                                                .cloned()
559                                                            {
560                                                                self.store
561                                                                    .linked_by
562                                                                    .insert(split, links);
563                                                            }
564                                                        }
565
566                                                        self.merge_blocks.push(*split.id());
567                                                        index += 1;
568                                                    }
569                                                }
570                                                self.delete(item);
571                                                blocks = self
572                                                    .store
573                                                    .blocks
574                                                    .get_client_mut(client)
575                                                    .unwrap();
576                                                // just to make the borrow checker happy
577                                            }
578                                        } else {
579                                            break;
580                                        }
581                                    }
582                                    index += 1;
583                                }
584                            }
585                        }
586                    } else {
587                        unapplied.insert(ID::new(*client, clock), clock_end - clock);
588                    }
589                }
590            }
591        }
592
593        if unapplied.is_empty() {
594            None
595        } else {
596            Some(unapplied)
597        }
598    }

600    /// Delete item under given pointer.
601    /// Returns true if block was successfully deleted, false if it was already deleted in the past.
602    pub(crate) fn delete(&mut self, mut item: ItemPtr) -> bool {
603        let mut recurse = Vec::new();
604        let mut result = false;
605
606        let ptr = item.clone();
607        let store = self.store.deref();
608        if !item.is_deleted() {
609            if item.parent_sub.is_none() && item.is_countable() {
610                if let TypePtr::Branch(mut parent) = item.parent {
611                    parent.block_len -= item.len();
612                    parent.content_len -= item.content_len(store.offset_kind);
613                }
614            }
615
616            item.mark_as_deleted();
617            self.delete_set.insert(item.id.clone(), item.len());
618            if let Some(parent) = item.parent.as_branch() {
619                self.add_changed_type(*parent, item.parent_sub.clone());
620            } else {
621                // parent has been GC'ed
622            }
623
624            match &mut item.content {
625                ItemContent::Doc(_, doc) => {
626                    let subdocs = self.subdocs.get_or_init();
627                    let addr = doc.addr();
628                    if subdocs.added.remove(&addr).is_none() {
629                        subdocs.removed.insert(addr, doc.clone());
630                    }
631                }
632                ItemContent::Type(inner) => {
633                    let branch_ptr = BranchPtr::from(inner);
634                    #[cfg(feature = "weak")]
635                    if let crate::types::TypeRef::WeakLink(source) = &branch_ptr.type_ref {
636                        source.unlink_all(self, branch_ptr);
637                    }
638                    let mut ptr = branch_ptr.start;
639                    self.changed.remove(&TypePtr::Branch(branch_ptr));
640
641                    while let Some(item) = ptr.as_deref() {
642                        if !item.is_deleted() {
643                            recurse.push(ptr.unwrap());
644                        }
645
646                        ptr = item.right.clone();
647                    }
648
649                    for ptr in branch_ptr.map.values() {
650                        recurse.push(ptr.clone());
651                    }
652                }
653                ItemContent::Move(m) => m.delete(self, ptr),
654                _ => { /* nothing to do for other content types */ }
655            }
656            if item.info.is_linked() {
657                // notify links that current element has been removed
658                if let Some(linked_by) = self.store.linked_by.remove(&item) {
659                    for link in linked_by {
660                        self.add_changed_type(link, item.parent_sub.clone());
661                        #[cfg(feature = "weak")]
662                        if let crate::types::TypeRef::WeakLink(source) = &link.type_ref {
663                            if source.is_single() {
664                                source.first_item.take();
665                            }
666                        }
667                    }
668                }
669            }
670            result = true;
671        }
672
673        for &ptr in recurse.iter() {
674            let id = *ptr.id();
675            if !self.delete(ptr) {
676                // Whis will be gc'd later and we want to merge it if possible
677                // We try to merge all deleted items after each transaction,
678                // but we have no knowledge about that this needs to be merged
679                // since it is not in transaction.ds. Hence we add it to transaction._mergeStructs
680                self.merge_blocks.push(id);
681            }
682        }
683
684        result
685    }

687    /// Applies a deserialized [Update] contents into a document owning current transaction. Update
688    /// payload can be generated by methods such as [TransactionMut::encode_diff] or passed to
689    /// [Doc::observe_update_v1]/[Doc::observe_update_v2] callbacks. Updates are allowed to contain
690    /// duplicate blocks (already presen in current document store) - these will be ignored.
691    ///
692    /// # Pending updates
693    ///
694    /// Remote update integration requires that all to-be-integrated blocks must have their direct
695    /// predecessors already in place. Out of order updates from the same peer will be stashed
696    /// internally and their integration will be postponed until missing blocks arrive first.
697    pub fn apply_update(&mut self, update: Update) -> Result<(), UpdateError> {
698        let (remaining, remaining_ds) = update.integrate(self)?;
699        let mut retry = false;
700        {
701            let store = self.store_mut();
702            store.pending = if let Some(mut pending) = store.pending.take() {
703                // check if we can apply something
704                for (client, &clock) in pending.missing.iter() {
705                    if clock < store.blocks.get_clock(client) {
706                        retry = true;
707                        break;
708                    }
709                }
710
711                if let Some(remaining) = remaining {
712                    // merge restStructs into store.pending
713                    for (&client, &clock) in remaining.missing.iter() {
714                        pending.missing.set_min(client, clock);
715                    }
716                    pending.update = Update::merge_updates(vec![pending.update, remaining.update]);
717                }
718                Some(pending)
719            } else {
720                remaining
721            };
722        }
723        if let Some(pending) = self.store_mut().pending_ds.take() {
724            let ds2 = self.apply_delete(&pending);
725            let ds = match (remaining_ds, ds2) {
726                (Some(mut a), Some(b)) => {
727                    a.delete_set.merge(b);
728                    Some(a.delete_set)
729                }
730                (Some(x), _) => Some(x.delete_set),
731                (_, Some(x)) => Some(x),
732                _ => None,
733            };
734            self.store_mut().pending_ds = ds;
735        } else {
736            self.store_mut().pending_ds = remaining_ds.map(|update| update.delete_set);
737        }
738
739        if retry {
740            let store = self.store_mut();
741            if let Some(pending) = store.pending.take() {
742                let ds = store.pending_ds.take().unwrap_or_default();
743                let mut ds_update = Update::new();
744                ds_update.delete_set = ds;
745                self.apply_update(pending.update)?;
746                self.apply_update(ds_update)?;
747            }
748        }
749
750        Ok(())
751    }

753    pub(crate) fn create_item<T: Prelim>(
754        &mut self,
755        pos: &ItemPosition,
756        value: T,
757        parent_sub: Option<Arc<str>>,
758    ) -> Option<ItemPtr> {
759        let (left, right, origin, id) = {
760            let store = self.store_mut();
761            let left = pos.left;
762            let right = pos.right;
763            let origin = if let Some(item) = pos.left.as_deref() {
764                Some(item.last_id())
765            } else {
766                None
767            };
768            let client_id = store.client_id;
769            let id = ID::new(client_id, store.get_local_state());
770
771            (left, right, origin, id)
772        };
773        let (mut content, remainder) = value.into_content(self);
774        let inner_ref = if let ItemContent::Type(inner_ref) = &mut content {
775            Some(BranchPtr::from(inner_ref))
776        } else {
777            None
778        };
779        let mut block = Item::new(
780            id,
781            left,
782            origin,
783            right,
784            right.map(|r| r.id().clone()),
785            pos.parent.clone(),
786            parent_sub,
787            content,
788        )?;
789        let mut block_ptr = ItemPtr::from(&mut block);
790
791        block_ptr.integrate(self, 0);
792
793        self.store_mut().blocks.push_block(block);
794
795        if let Some(remainder) = remainder {
796            remainder.integrate(self, inner_ref.unwrap().into())
797        }
798
799        Some(block_ptr)
800    }

802    fn call_type_observers(
803        changed_parent_types: &mut Vec<BranchPtr>,
804        all_links: &HashMap<ItemPtr, HashSet<BranchPtr>>,
805        branch: BranchPtr,
806        changed_parents: &mut HashMap<BranchPtr, Vec<usize>>,
807        event_cache: &Vec<Event>,
808        visited: &mut HashSet<BranchPtr>,
809    ) {
810        let mut current = branch;
811        loop {
812            changed_parent_types.push(current);
813            if current.deep_observers.has_subscribers() {
814                let entries = changed_parents.entry(current).or_default();
815                entries.push(event_cache.len() - 1);
816            }
817
818            if let Some(item) = current.item {
819                if item.info.is_linked() {
820                    if let Some(linked_by) = all_links.get(&item) {
821                        for &link in linked_by.iter() {
822                            if visited.insert(link) {
823                                Self::call_type_observers(
824                                    changed_parent_types,
825                                    all_links,
826                                    link,
827                                    changed_parents,
828                                    event_cache,
829                                    visited,
830                                )
831                            }
832                        }
833                    }
834                }
835                if let TypePtr::Branch(parent) = item.parent {
836                    current = parent;
837                    continue;
838                }
839            }
840
841            break;
842        }
843    }

845    /// Commits current transaction. This step involves cleaning up and optimizing changes performed
846    /// during lifetime of a transaction. Such changes include squashing delete sets data,
847    /// squashing blocks that have been appended one after another to preserve memory and triggering
848    /// events.
849    ///
850    /// This step is performed automatically when a transaction is about to be dropped (its life
851    /// scope comes to an end).
852    pub fn commit(&mut self) {
853        if self.committed {
854            return;
855        }
856        self.committed = true;
857
858        // 1. sort and merge delete set
859        self.delete_set.squash();
860        self.after_state = self.store.blocks.get_state_vector();
861        // 2. emit 'beforeObserverCalls'
862        // 3. for each change observed by the transaction call 'afterTransaction'
863        if !self.changed.is_empty() {
864            let mut changed_parents: HashMap<BranchPtr, Vec<usize>> = HashMap::new();
865            let mut event_cache = Vec::new();
866
867            for (ptr, subs) in self.changed.iter() {
868                if let TypePtr::Branch(branch) = ptr {
869                    if let Some(e) = branch.trigger(self, subs.clone()) {
870                        event_cache.push(e);
871                        Self::call_type_observers(
872                            &mut self.changed_parent_types,
873                            &self.store.linked_by,
874                            *branch,
875                            &mut changed_parents,
876                            &event_cache,
877                            &mut HashSet::default(),
878                        );
879                    }
880                }
881            }
882
883            // deep observe events
884            for (&branch, events) in changed_parents.iter() {
885                // sort events by path length so that top-level events are fired first.
886                let mut unsorted: Vec<&Event> = Vec::with_capacity(events.len());
887
888                for &i in events.iter() {
889                    let e = &mut event_cache[i];
890                    e.set_current_target(branch);
891                }
892
893                for &i in events.iter() {
894                    unsorted.push(&event_cache[i]);
895                }
896
897                // We don't need to check for events.length
898                // because we know it has at least one element
899                let events = Events::new(&mut unsorted);
900                branch.trigger_deep(self, &events);
901            }
902        }
903
904        if let Some(events) = self.store.events.take() {
905            events.emit_after_transaction(self);
906            self.store.events = Some(events);
907        }
908
909        // 4. try GC delete set
910        if !self.store.skip_gc {
911            GCCollector::collect(self);
912        }
913
914        // 5. try merge delete set
915        self.delete_set.try_squash_with(&mut self.store);
916
917        // 6. get transaction after state and try to merge to left
918        for (client, &clock) in self.after_state.iter() {
919            let before_clock = self.before_state.get(client);
920            if before_clock != clock {
921                let blocks = self.store.blocks.get_client_mut(client).unwrap();
922                let first_change = blocks.find_pivot(before_clock).unwrap().max(1);
923                let mut i = blocks.len() - 1;
924                while i >= first_change {
925                    blocks.squash_left(i);
926                    i -= 1;
927                }
928            }
929        }
930
931        // 7. get merge_structs and try to merge to left
932        for id in self.merge_blocks.iter() {
933            if let Some(blocks) = self.store.blocks.get_client_mut(&id.client) {
934                if let Some(replaced_pos) = blocks.find_pivot(id.clock) {
935                    if replaced_pos + 1 < blocks.len() {
936                        blocks.squash_left(replaced_pos + 1);
937                    } else if replaced_pos > 0 {
938                        blocks.squash_left(replaced_pos);
939                    }
940                }
941            }
942        }
943
944        if let Some(events) = self.store.events.as_ref() {
945            // 8. emit 'afterTransactionCleanup'
946            events.emit_transaction_cleanup(self);
947            // 9. emit 'update'
948            events.emit_update_v1(self);
949            // 10. emit 'updateV2'
950            events.emit_update_v2(self);
951        }
952
953        // 11. add and remove subdocs
954        let store = self.store.deref_mut();
955        if let Some(mut subdocs) = self.subdocs.take() {
956            let client_id = store.client_id;
957            for (guid, subdoc) in subdocs.added.iter_mut() {
958                let mut txn = subdoc.transact_mut();
959                txn.store.client_id = client_id;
960                txn.doc
961                    .store()
962                    .set_subdoc_data(client_id, self.doc.collection_id());
963                store.subdocs.insert(guid.clone(), subdoc.clone());
964            }
965            for guid in subdocs.removed.keys() {
966                store.subdocs.remove(guid);
967            }
968
969            let store = self.store.deref();
970            let mut removed = if let Some(events) = store.events.as_ref() {
971                if events.subdocs_events.has_subscribers() {
972                    let e = SubdocsEvent::new(subdocs);
973                    events.subdocs_events.trigger(|cb| cb(self, &e));
974                    e.removed
975                } else {
976                    subdocs.removed
977                }
978            } else {
979                subdocs.removed
980            };
981
982            for (_, subdoc) in removed.iter_mut() {
983                subdoc.destroy(self);
984            }
985        }
986    }

996        let trigger = if let Some(ptr) = parent.item {
997            (ptr.id().clock < self.before_state.get(&ptr.id().client)) && !ptr.is_deleted()
998        } else {
999            true
1000        };

1003            e.insert(parent_sub.clone());
1004        }

1017    pub(crate) fn split_by_snapshot(&mut self, snapshot: &Snapshot) {
1018        let mut merge_blocks: Vec<ID> = Vec::new();
1019        let blocks = &mut self.store.blocks;
1020        for (&client, &clock) in snapshot.state_map.iter() {
1021            if let Some(ptr) = blocks.get_item(&ID::new(client, clock)) {
1022                let ptr_clock = ptr.id.clock;
1023                if ptr_clock < clock {
1024                    if let Some(right) = blocks.split_block_inner(ptr, clock - ptr_clock) {
1025                        if right.moved.is_some() {
1026                            if let Some(&prev_moved) = self.prev_moved.get(&ptr) {
1027                                self.prev_moved.insert(right, prev_moved);
1028                            }
1029                        }
1030
1031                        merge_blocks.push(*right.id());
1032                    }
1033                }
1034            }
1035        }
1036
1037        self.merge_blocks.append(&mut merge_blocks);
1038        let mut deleted = snapshot.delete_set.deleted_blocks();
1039        while let Some(slice) = deleted.next(self) {
1040            if let BlockSlice::Item(slice) = slice {
1041                //TODO: we technically don't need to physically split underlying item in two
1042                // if we were to use block slices all the way down.
1043
1044                // split the blocks by delete set
1045                let ptr = self.store.materialize(slice);
1046                self.merge_blocks.push(ptr.id);
1047            }
1048        }
1049    }

1058    #[cfg(feature = "weak")]
1059    pub(crate) fn unlink(&mut self, mut source: ItemPtr, link: BranchPtr) {
1060        let all_links = &mut self.store.linked_by;
1061        let prune = if let Some(linked_by) = all_links.get_mut(&source) {
1062            linked_by.remove(&link) && linked_by.is_empty()
1063        } else {
1064            false
1065        };
1066        if prune {
1067            all_links.remove(&source);
1068            source.info.clear_linked();
1069            if source.is_countable() {
1070                // since linked property is blocking items from merging,
1071                // it may turn out that source item can be merged now
1072                self.merge_blocks.push(source.id);
1073            }
1074        }
1075    }
